import math
radius=5
PI=math.pi
area=PI*radius**2
circumference=2*PI*radius
print(f"Area: {area}, circumference: {circumference}")



temp_celsius=int(input("Write temperature in Celsius: "))
temp_kelvin=temp_celsius+273.15
temp_fahrenheit=temp_celsius*1.8+32
print(f"Celsius: {temp_celsius}, Kelvin: {temp_kelvin}, Fahrenheit: {temp_fahrenheit}")
